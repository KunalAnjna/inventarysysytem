<script>
    // Initialize Data
    let inventory = JSON.parse(localStorage.getItem('toolInventory')) || [];

    // Save to LocalStorage
    function saveData() {
        localStorage.setItem('toolInventory', JSON.stringify(inventory));
        updateStats();
        renderTable();
    }

    // Add Tool
    function addTool() {
        const name = document.getElementById('toolName').value;
        const category = document.getElementById('toolCategory').value;
        const serial = document.getElementById('toolSerial').value;
        const user = document.getElementById('toolUser').value;

        if (!name || !serial) {
            alert("Please enter at least a Name and Serial Number.");
            return;
        }

        const newTool = {
            id: Date.now(), // Unique ID based on timestamp
            name: name,
            category: category,
            serial: serial,
            user: user || 'N/A',
            status: user ? 'checked-out' : 'available',
            dateAdded: new Date().toLocaleDateString()
        };

        inventory.push(newTool);
        saveData();
        
        // Clear inputs
        document.getElementById('toolName').value = '';
        document.getElementById('toolSerial').value = '';
        document.getElementById('toolUser').value = '';
    }

    // Toggle Status (Check In / Check Out)
    function toggleStatus(id) {
        const tool = inventory.find(t => t.id === id);
        if (tool) {
            if (tool.status === 'available') {
                const user = prompt("Enter user name to check out:");
                if (user) {
                    tool.status = 'checked-out';
                    tool.user = user;
                }
            } else {
                tool.status = 'available';
                tool.user = 'N/A';
            }
            saveData();
        }
    }

    // Delete Tool
    function deleteTool(id) {
        if (confirm("Are you sure you want to delete this tool?")) {
            inventory = inventory.filter(t => t.id !== id);
            saveData();
        }
    }

    // Render Table
    function renderTable() {
        const tbody = document.querySelector('#inventoryTable tbody');
        const search = document.getElementById('searchInput').value.toLowerCase();
        tbody.innerHTML = '';

        inventory.forEach(tool => {
            if (tool.name.toLowerCase().includes(search) || 
                tool.serial.toLowerCase().includes(search) || 
                tool.user.toLowerCase().includes(search)) {
                
                const tr = document.createElement('tr');
                tr.innerHTML = `
                    <td>${tool.id}</td>
                    <td><strong>${tool.name}</strong></td>
                    <td>${tool.category}</td>
                    <td>${tool.serial}</td>
                    <td><span class="status ${tool.status}">${tool.status.toUpperCase()}</span></td>
                    <td>${tool.user}</td>
                    <td>
                        <button onclick="toggleStatus(${tool.id})">${tool.status === 'available' ? 'Check Out' : 'Check In'}</button>
                        <button class="delete" onclick="deleteTool(${tool.id})">Delete</button>
                    </td>
                `;
                tbody.appendChild(tr);
            }
        });
    }

    // Update Statistics
    function updateStats() {
        document.getElementById('total-tools').innerText = inventory.length;
        document.getElementById('checked-out').innerText = inventory.filter(t => t.status === 'checked-out').length;
        document.getElementById('available').innerText = inventory.filter(t => t.status === 'available').length;
    }

    // Export to CSV
    function exportToCSV() {
        let csvContent = "data:text/csv;charset=utf-8,";
        csvContent += "ID,Name,Category,Serial,Status,User,Date Added\n";
        
        inventory.forEach(row => {
            csvContent += `${row.id},${row.name},${row.category},${row.serial},${row.status},${row.user},${row.dateAdded}\n`;
        });

        const encodedUri = encodeURI(csvContent);
        const link = document.createElement("a");
        link.setAttribute("href", encodedUri);
        link.setAttribute("download", "tool_inventory.csv");
        document.body.appendChild(link);
        link.click();
        document.body.removeChild(link);
    }

    // Initial Load
    renderTable();
    updateStats();
</script>
